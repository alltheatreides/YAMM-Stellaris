	entity = "empty_section_entity"
	construction_entity = "construction_platform_entity"
	portrait = "GFX_megastructure_construction_background"
	place_entity_on_planet_plane = yes
	entity_offset = { x = 0 y = 0 }
	build_time = 1000
	build_cost = {
		minerals = 8000
		influence = 100
	}
	construction_blocks_others = no

	potential = {		
		has_ascension_perk = ap_voidborn
	}

	possible = {
		custom_tooltip = {
			fail_text = "requires_inside_border"
			is_inside_border = from
		}
		
	}

	placement_rules = {
		planet_possible = {
			custom_tooltip = {
				fail_text = "requires_no_anomaly"
				NOT = { has_anomaly = yes }
			}
			custom_tooltip = {
				fail_text = "Planet is already an Ecumenopolis."
				NOT = { has_planet_flag = ecumenopolis }
			}		
			
			custom_tooltip = {
				fail_text = "requires_no_orbital_station"
				has_orbital_station = no
			}
			custom_tooltip = {
				fail_text = "requires_survey_not_habitable"
				is_surveyed = {			# prevent leaking habitability information
					who = prev.from
					status = yes
				}
			}
			custom_tooltip = {
				fail_text = "Can't build around habitat or asteroid."
				has_owner = yes
				NOT = {
					OR = {
						is_planet_class = "pc_habitat"						
						is_asteroid = yes
					}	
				}
			}
			custom_tooltip = {
				fail_text = "requires_not_star"
				is_star = no
			}
			
			custom_tooltip = {
				fail_text = "requires_planetary_capital"
				OR = { 
					has_building = building_capital_2
					has_building = building_capital_3
				}
			}
			
		} # use these for all non-star megastructures
	}
	
	ai_weight = {
		modifier = {
			weight = 0
			NOT = {
			}
		}
	}
	
	on_build_start = {}
	on_build_cancel = {}	
	on_build_complete = {
		fromfrom.planet = {		    
			add_modifier = { modifier = ecumenopolis_under_construction }
			set_planet_flag = ecumenopolis_under_construction
			remove_planet_flag = prison_world
			remove_planet_flag = research_world
			remove_planet_flag = fortress_world
			remove_planet_flag = agri_world
			remove_planet_flag = trade_world
			remove_planet_flag = recruiting_world	
			remove_planet_flag = mining_world
			remove_planet_flag = outpost_world
			remove_planet_flag = cultural_world
			remove_planet_flag = forge_world
			remove_planet_flag = diplomatic_world
			remove_modifier = mining_world
			remove_modifier = prison_world
			remove_modifier = recruiting_world
			remove_modifier = research_world
			remove_modifier = fortress_world
			remove_modifier = agri_world
			remove_modifier = trade_world
			remove_modifier = outpost_world
			remove_modifier = cultural_world
			remove_modifier = forge_world
			remove_modifier = diplomatic_world
				
		}
		
	}
	
}

ecumenopolis_3 = {
	entity = "empty_section_entity"
	construction_entity = "empty_section_entity"
	portrait = "GFX_megastructure_construction_background"
	upgrade_from = {
		ecumenopolis_2
	}
	build_time = 600
	build_cost = {
		minerals = 2000
	}	
	
	on_build_complete = {
		fromfrom.planet = {
		    change_pc = pc_ecumenopolis
			remove_modifier = ecumenopolis_under_construction
			remove_planet_flag = ecumenopolis_under_construction
			set_planet_flag = ecumenopolis
			add_modifier = { modifier = ecumenopolis }
			
			remove_planet = no 			
		}
		owner = {
			add_modifier = { modifier = ecumenopolis_empire }
			add_monthly_resource_mult = {
				resource = influence
				value = 150				
			}
		}
		remove_megastructure = fromfrom
	}
}
